    extend = function (target, source) {
    target = target || {};
    source = source || {};

    for (var prop in source) {
        /* istanbul ignore else */
        if (source.hasOwnProperty(prop)) {
            if (source[prop] instanceof Array) {
                // don't edit Array object - just replace
                target[prop] = source[prop];
            } else if (typeof source[prop] === 'object') {
                target[prop] = extend(target[prop], source[prop]);
            } else {
                target[prop] = source[prop];
            }
        }
    }
    return target;
}

function node-config-manager.object.isEmpty (obj)

  • description and source-code
    isEmpty = function (obj) {
    // null and undefined are "empty"
    if (obj == null) {
        return true;
    }

    // Assume if it has a length property with a non-zero value
    // that that property is correct.
    if (obj.length > 0) {
        return false;
    }
    if (obj.length === 0) {
        return true;
    }

    // Otherwise, does it have any properties of its own?
    // Note that this doesn't handle
    // toString and valueOf enumeration bugs in IE < 9
    for (var key in obj) {
        /* istanbul ignore else */
        if (obj.hasOwnProperty(key)) {
            return false;
        }
    }

    return true;
}

function node-config-manager.parameter.parse (json, camelCase)

  • description and source-code
    parse = function (json, camelCase) {
    if ('object' !== typeof json) {
        return null;
    }

    camelCase = camelCase || false;

    //Get keys
    var levels, value, attr, _pObject, pObject,
        object = {},
        keys = Object.keys(json);

    keys.forEach(function(key) {
        value = json[key];

        //split levels
        levels = key.split('__');

        //Back to top
        _pObject = pObject = object;
        attr = null;

        levels.forEach(function(level) {
            attr = convertStrArrayToString(level.split('_'), camelCase);

            //Persist
            _pObject = pObject;

            //New Object
            pObject[attr] = pObject[attr] || {};
            pObject = pObject[attr];
        });

        //Save value
        /* istanbul ignore else */
        if (_pObject && attr) {
            _pObject[attr] = validateValue(processTag(value));
        }
    });

    return object;
}

  • description and source-code
    processTag = function (str) {
    if ('string' !== typeof str) {
        return str;
    }

    var tag;
    while (str.search(tagRegex) >= 0) {
        tag = str.match(tagRegex)[0];
        str = str.replace(tag, resolveTag(tag), 'g');
    }
    return str;
}
